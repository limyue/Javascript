event.stopPropagation();阻止事件冒泡 

     window.onload = function(){
      var btn = document.querySelector('button');
      var story = document.querySelector('.story');
      var customname = document.querySelector('.customname');


      var arr1 = [
          '今天早上，在学校',
          '今天中午，在公交车上',
          '今天傍晚，在公园里'
      ]
      var arr2 = [
          ',偷偷放屁',
          ',明目张胆的吃零食',
          ',乱扔垃圾，被打了一顿'
      ]

      btn.addEventListener('click',runStory);
      function runStory(){
        //console.log('点击按钮生效')
        
        var num1 = Math.floor(Math.random()*3);
        var num2 = Math.floor(Math.random()*3);
       
        story.style.visibility = 'visible';
        story.textContent = '故事';
        story.textContent = arr1[num1]+customname.value+arr2[num2];

      }
     }
